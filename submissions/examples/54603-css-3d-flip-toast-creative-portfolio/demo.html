Implementation Quick Start

CSS Usage
style.css & demo.html 
/* 1. 3D Flip Entrance Keyframe Animation */
@keyframes em3DFlipEntrance {
  0%   { opacity: 0; transform: perspective(1200px) rotateX(-90deg) translateZ(-50px); }
  65%  { opacity: 1; transform: perspective(1200px) rotateX(12deg) translateZ(10px); }
  100% { opacity: 1; transform: perspective(1200px) rotateX(0deg) translateZ(0); }
}

/* 2. 3D Tilt Hover Rule */
.em-3d-flip-toast:hover {
  transform: perspective(1000px) rotateX(6deg) rotateY(-8deg) translateZ(15px);
  box-shadow: 0 35px 70px -15px rgba(0, 0, 0, 0.85), 0 0 40px var(--toast-glow-color);
}

/* 3. HTML Markup Structure */
<div class="em-toast-container toast-pos-top-right">
  <div class="em-3d-flip-toast toast-variant-success toast-show" role="status" aria-live="polite">
    <div class="toast-body-flex">
      <div class="toast-icon-wrapper">...</div>
      <div class="toast-details">
        <h4 class="toast-title">3D Scene Published!</h4>
        <p class="toast-message">High-res render live on portfolio.</p>
      </div>
    </div>
  </div>
</div>
