Usage

<nav aria-label="Pagination">
  <ul class="ease-pagination">
    <li class="is-disabled"><span>&laquo; Prev</span></li>
    <li class="is-active"><span>1</span></li>
    <li><a href="#">2</a></li>
    <li><a href="#">3</a></li>
    <li><a href="#">Next &raquo;</a></li>
  </ul>
</nav>

<!-- Sizes: .ease-pagination-sm, .ease-pagination-lg -->
<!-- Simple: .ease-pagination-simple (Prev / Page X of Y / Next) -->

<!-- is-active — highlight current page -->
<!-- is-disabled — gray out unavailable actions -->
