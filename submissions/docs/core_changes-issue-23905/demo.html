Usage

<!-- Basic chip -->
<span class="ease-chip ease-chip-primary">Chip</span>

<!-- Outline pill -->
<span class="ease-chip ease-chip-outline ease-chip-success ease-chip-pill">
  Tag
</span>

<!-- Removable -->
<span class="ease-chip ease-chip-info ease-chip-pill">
  Label
  <button class="ease-chip-remove" onclick="this.parentElement.remove()"
    aria-label="Remove">&#10005;</button>
</span>

<!-- Interactive / clickable -->
<button class="ease-chip ease-chip-primary ease-chip-interactive">
  Clickable
</button>

<!-- Chip group -->
<div class="ease-chip-group">
  <span class="ease-chip-group-label">Skills:</span>
  <span class="ease-chip ease-chip-primary">Design</span>
  <span class="ease-chip ease-chip-success">Dev</span>
</div>
