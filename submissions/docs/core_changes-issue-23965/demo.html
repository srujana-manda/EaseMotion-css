Usage

<!-- Create a counter list --> <ol class="ease-counter-list"> <li class="ease-counter-item">First item</li> <li class="ease-counter-item">Second item</li> </ol> /* Customize via CSS variables */ .my-counter { --ease-counter-prefix: '#'; --ease-counter-suffix: '. '; --ease-counter-color: #34d399; --ease-counter-weight: 800; }
