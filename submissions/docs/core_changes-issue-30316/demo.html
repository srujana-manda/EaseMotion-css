How It Works

/* Variable on :root */
:root { --ease-motion-reduce: 0; }

/* Auto-detect user preference */
@media (prefers-reduced-motion: reduce) {
  :root { --ease-motion-reduce: 1; }
}

/* Class for per-container reduction */
.ease-reduce-motion * { animation: none !important; transition: none !important; }

/* Manual toggle via JS (optional) */
document.documentElement.style.setProperty('--ease-motion-reduce', '1');
