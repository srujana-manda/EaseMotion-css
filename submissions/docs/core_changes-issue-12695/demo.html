ease-breadcrumb

Usage

<nav aria-label="Breadcrumb">
  <ol class="ease-breadcrumb ease-breadcrumb-chevron">
    <li><a href="#">Home</a></li>
    <li><a href="#">Products</a></li>
    <li<a href="#">Category</a></li>
    <li aria-current="page">Current Page</li>
  </ol>
</nav>
